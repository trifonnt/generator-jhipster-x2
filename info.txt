= Steps when installing generator-jhipster
$ cd /workspace
$ mkdir marketplace
$ cd marketplace
$ npm link generator-jhipster

$ npm link generator-jhipster-entity-audit
$ sudo npm install -g generator-jhipster-entity-audit

# Start JHipster generator
$ jhipster


= Steps when installing generator-jhipster-x2
$ sudo npm install -g generator-jhipster-x2
## Show which version is installed
$ jhipster-x2 --version


= Steps when publishing new version
## modify package.json
	"version": "6.9.1-a01",

$ git commit -m "Version ..."
$ git push --set-upstream origin v6.9.1-axx
$ git push

$ npm login ## ONE time only:
$ npm version 6.9.1-a01 --allow-same-version
$ npm publish
